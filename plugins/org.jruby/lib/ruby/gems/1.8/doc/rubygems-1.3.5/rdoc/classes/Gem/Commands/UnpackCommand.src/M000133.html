  def initialize
    super 'unpack', 'Unpack an installed gem to the current directory',
          :version => Gem::Requirement.default,
          :target  => Dir.pwd

    add_option('--target=DIR', 'target directory for unpacking') do |value, options|
      options[:target] = value
    end

    add_version_option
  end
