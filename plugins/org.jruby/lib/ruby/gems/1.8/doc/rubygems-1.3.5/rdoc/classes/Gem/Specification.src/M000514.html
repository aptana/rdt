  def _dump(limit)

      @summary,
      @required_ruby_version,
      @required_rubygems_version,
      @original_platform,
      @dependencies,
      @rubyforge_project,
      @email,
      @authors,
      @description,
      @homepage,
      @has_rdoc,
      @new_platform,
      @licenses
    ]
  end
