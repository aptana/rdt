  def self.array_attribute(name)
    @@non_nil_attributes << ["@#{name}".intern, []]

    @@array_attributes << name
    @@attributes << [name, []]
    @@default_value[name] = []
    code = %{
      def #{name}
        @#{name} ||= []
      end
      def #{name}=(value)
        @#{name} = Array(value)
      end
    }

    module_eval code, __FILE__, __LINE__ - 9
  end
