  def tsort_each_child(node, &block)
    specs = @specs.sort.reverse

    node.dependencies.each do |dep|
      specs.each do |spec|
        if spec.satisfies_requirement? dep then
          begin
            yield spec
          rescue TSort::Cyclic
          end
          break
        end
      end
    end
  end
