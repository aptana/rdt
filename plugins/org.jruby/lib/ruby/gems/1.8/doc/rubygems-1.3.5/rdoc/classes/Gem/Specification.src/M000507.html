  def self.attribute(name, default=nil)
    ivar_name = "@#{name}".intern
    if default.nil? then
      @@nil_attributes << ivar_name
    else
      @@non_nil_attributes << [ivar_name, default]
    end

    @@attributes << [name, default]
    @@default_value[name] = default
    attr_accessor(name)
  end
