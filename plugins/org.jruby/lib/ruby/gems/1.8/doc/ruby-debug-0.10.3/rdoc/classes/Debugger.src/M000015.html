    def info_thread(*args)
      unless args[0]
        info_threads(args[0])
        return
      end
      ok, verbose = info_thread_preamble(args[1])
      return unless ok
      c = parse_thread_num("info thread" , args[0])
      return unless c
      display_context(c, !verbose)
      if verbose and not c.ignored?
        (0...c.stack_size).each do |idx|
          print "\t"
          print_frame(idx, false, c) 
        end
      end
    end
