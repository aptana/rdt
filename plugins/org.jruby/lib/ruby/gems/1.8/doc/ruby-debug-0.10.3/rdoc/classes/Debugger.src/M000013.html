    def info_stack(*args)
      if not @state.context
        errmsg "info stack not available here.\n"
        return
      end
      (0...@state.context.stack_size).each do |idx|
        if idx == @state.frame_pos
          print "--> "
        else
          print "    "
        end
        print_frame(idx)
      end
    end
