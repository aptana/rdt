    def execute
      (0...@state.context.stack_size).each do |idx|
        if idx == @state.frame_pos
          print "--> "
        else
          print "    "
        end
        print_frame(idx)

      end
      if truncated_callstack?(@state.context, Debugger.start_sentinal)
        print "Warning: saved frames may be incomplete; compare with caller(0).\n" 
      end
    end
